Bootstrapping Application With Standalone Components

Standalone Components also provide ability to Bootstrapping Application. 

        {{"
            // in the main.ts file
            import {bootstrapApplication} from '@angular/platform-browser';
            import {StandaloneComponent} from './app/standalone.component';
            bootstrapApplication(StandaloneComponent);
        "}}

Configure values and services in bootstrapApplication 

        {{
            "
                bootstrapApplication(StandaloneComponent, {
                    providers: [
                        Service,
                        {provide: BACKEND_URL, useValue: 'https://ul.com'},
                        // ...
                    ]
              });
            "
        }}

Configure Dependencies which are dependent on NgModule 

        {{"
            bootstrapApplication(PhotoAppComponent, {
                providers: [
                    {provide: BACKEND_URL, useValue: 'https://ul.com'},
                    importProvidersFrom(
                        RouterModule.forRoot([/* app routes */]),
                    ),
                    // ...
                ]
            });
        "}}
